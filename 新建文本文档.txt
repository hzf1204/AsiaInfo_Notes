create or replace procedure myprocedure2(  hh in number)
IS
       the_no number(3):=hh;
       cursor mycur(hhh number) is(select * from student a where a.no=hhh);
       myrec mycur%rowtype;
       
       begin
           open mycur(the_no);
               fetch mycur into myrec;
               if mycur%found then
                   DBMS_OUTPUT.PUT_LINE('HELLO');
                   delete from student a where a.no=the_no);
                   commit;
               end if;
           close mycur;
       end myprocedure2;
       
begin           
    myprocedure2(5);
end;
